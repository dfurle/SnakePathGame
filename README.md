# Variables

- PLAY_MANUAL = false;
    - play manually = true
    - let path finder play = false
- FRAMES = 60;
    - how fast to run the path finder, in FPS.
    - manual mode auto sets to 10 FPS inside Game Constructor
- COLS = 30;
    - columns in board
- ROWS = 20;
    - rows in board
- Block::size = 30;
    - size of each block in px
